create or replace PROCEDURE PHB_APPROVE_REJECT_REPORT
(
  MA_BAO_CAO IN NVARCHAR2,
  REFID_BC IN NVARCHAR2,
  TRANG_THAI IN NUMBER
)AS
QUERY_STR  VARCHAR2(300); 
cur SYS_REFCURSOR;
BEGIN
QUERY_STR:='UPDATE '||MA_BAO_CAO||' SET TRANG_THAI='||TRANG_THAI||' WHERE REFID='''||REFID_BC||'''';
DBMS_OUTPUT.PUT_LINE(QUERY_STR);
EXECUTE IMMEDIATE QUERY_STR;
END PHB_APPROVE_REJECT_REPORT;
