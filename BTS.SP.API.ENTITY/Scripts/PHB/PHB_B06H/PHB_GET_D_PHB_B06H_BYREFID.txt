create or replace PROCEDURE PHB_GET_D_PHB_B06H_BYREFID
(
  REF_ID IN NVARCHAR2,
  cur OUT SYS_REFCURSOR
)AS
BEGIN
Open cur for 
SELECT B06HD.*
FROM PHB_B06H B06H,PHB_B06H_DETAIL B06HD
WHERE   B06H.REFID=REF_ID AND B06H.REFID=B06HD.PHB_B06H_REFID ORDER BY PHAN,MA_LOAI,MA_KHOAN,MA_CHI_TIEU;
END PHB_GET_D_PHB_B06H_BYREFID;
